191/**
192 * @brief ポート制御
193 *
194 * @param number ON/OFFするモータ番号
195 */
196void OutputPort(int number)
197{
198 if(motor_state[number][2] == 0)
199 {
200 digitalWrite(motor_state[number][0],LOW);
201 time_count[number] = 0;
202 motor_state[number][1] = 0;
203 }else{
204 if(motor_state[number][1] == 1)
205 {
206 if(time_count[number] >= motor_state[number][2])
207 {
208 digitalWrite(motor_state[number][0],LOW);
209 time_count[number] = 0;
210 motor_state[number][1] = 0;
211 }
212 }else{
213 if(time_count[number] >= (motor_period - motor_state[number][2]))
214 {
215 digitalWrite(motor_state[number][0],HIGH);
216 time_count[number] = 0;
217 motor_state[number][1] = 1;
218 }
219 }
220 }
221}

347/**
348 * @brief 文字列を区切り文字にて分割する
349 *
350 * @param data 文字列
351 * @param delimiter 区切り文字
352 * @param dst 格納先
353 * @return int
354 */
355int split(String data , char delimiter , String *dst)
356{
357 int index = 0;
358 int ArraySize = (sizeof(data) / sizeof((data)[0]));
359 int datalength = data.length();
360 for(int i = 0; i < datalength; i++)
361 {
362 char tmp = data.charAt(i);
363 if(tmp == delimiter)
364 {
365 index++;
366 if(index > (ArraySize - 1)) return -1;
367 }
368 else dst[index] += tmp;
369 }
370 return (index + 1);
371}

373/**
374 * @brief 受信したコマンドに応じた処理
375 *
376 * @param serial_data
377 */
378void SerialCheck(String serial_data)
379{
380 char const* const set_config[ ] = { "MOTOR",
381 "ALLMOTOR",
382 "PRM",
383 "VALVE",
384 "MOD",
385 "STOP",
386 "COMP"};
387
388 // 区切り文字「:」で入力された文字列を分割する
389 int index = split(serial_data, ':', g_cmds);
390
391 for(int i = 0; i < index; i++)
392 {
393 g_data[i] = g_cmds[i];
394 }
395
396 // 配列のクリア
397 for(unsigned int i = 0; i < ARRAY_SIZE(g_cmds); i++)
398 {
399 g_cmds[i] = '\0';
400 }
401
402 // 入力と一致するコマンドを探す
403 for(int i = 0; i < Command_MaxNumber; i++)
404 {
405 select_config = g_data[0].equalsIgnoreCase(String(set_config[i]));
406 if(select_config == 1)
407 {
408 config_number = number_count;
409 break;
410 }
411 number_count++;
412 }
413
414 // 一致しなかったとき
415 if(number_count == Command_MaxNumber)
416 {
417 config_number = 99;
418 }
419 DebugVariable("config_number",config_number);
420
421 // コマンド毎の処理
422 switch (config_number)
423 {
424 case Command_motor:
425 {
426 long set_motor_num = g_data[1].toInt();
427 long receive_duty = g_data[2].toInt();
428
429 if((receive_duty < Max_Duty) && (set_motor_num < Number_of_motor))
430 {
431 motor_state[set_motor_num][2] = receive_duty;
432 sprintf(send_text,
433 "Motor No: %ld\nSet Duty: %ld%%",
434 set_motor_num,
435 motor_state[set_motor_num][2]);
436 Serial.println(send_text);
437 }else{
438 Serial.println(Error1);
439 }
440 }
441 break;
442
443 case Command_allmotor:
444 {
445 long receive_duty = g_data[1].toInt();
446
447 if(receive_duty < Max_Duty)
448 {
449 for(unsigned int i = 0; i < Number_of_motor; i++)
450 {
451 motor_state[i][2] = receive_duty;
452 }
453
454 sprintf(send_text,"Set Duty: %ld%%",receive_duty);
455 Serial.println(send_text);
456 }else{
457 Serial.println(Error1);
458 }
459 }
460 break;
461
462 case Command_prm:
463 {
464 long set_motor_num = g_data[1].toInt();
465
466 if(set_motor_num < Number_of_motor)
467 {
468 sprintf(send_text,
469 "Motor No: %ld Duty: %ld%%",
470 set_motor_num,
471 motor_state[set_motor_num][2]);
472 Serial.println(send_text);
473 }else{
474 Serial.println(Error1);
475 }
476 }
477 break;
478
479 case Command_valve:
480 {
481 long set_valve_num = g_data[1].toInt();
482 long set_valve_state = g_data[2].toInt();
483
484 if(set_valve_num < Number_of_valves)
485 {
486 Valve_Fire(set_valve_num,set_valve_state);
487
488 if(set_valve_state)
489 {
490 sprintf(send_text,"Valve No: %ld ON",set_valve_num);
491 Serial.println(send_text);
492 }else{
493 sprintf(send_text,"Valve No: %ld OFF",set_valve_num);
494 Serial.println(send_text);
495 }
496 }else{
497 Serial.println(Error1);
498 }
499 }
500 break;
501
502 case Command_mod:
503 {
504 long receive_state = g_data[1].toInt();
505
506 if(receive_state < 3)
507 {
508 press_set_flag = receive_state;
509 if(press_set_flag)
510 {
511 Serial.println("SW Enable");
512 }else{
513 Serial.println("SW Disenable");
514 }
515 }else{
516 Serial.println(Error1);
517 }
518 }
519 break;
520
521 case Command_stop:
522 {
523 long receive_stop_press = g_data[1].toInt();
524
525 if((receive_stop_press < Max_Presser) && (Min_Presser < receive_stop_press))
526 {
527 sprintf(send_text,"@0,W-P_STOP,%ld",receive_stop_press);
528 Send_Data(send_text);
529 sprintf(send_text,"Set %ld[kPa]",receive_stop_press);
530 Serial.println(send_text);
531 }else{
532 Serial.println(Error1);
533 }
534 }
535 break;
536
537 default:
538 Serial.println(Error2);
539 break;
540 }
541
542 // 各種変数の初期化
543 for(int i = 0; i < 3; i++)
544 {
545 g_data[i] = '\0';
546 }
547 config_number = 0;
548 select_config = 0;
549 number_count = 0;
550}
